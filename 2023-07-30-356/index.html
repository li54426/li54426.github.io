        unordered_map<int, int> mp, rem;

            rem[n]++;
        }

        int res = 0;
        
        while(r< len){
            int n = nums[r++];
            mp[n]++;
            
            while(mp.size() == rem.size() && l< r){
                // åœ¨å·¦ç«¯ç‚¹çš„æƒ…å†µä¸‹
                // æšä¸¾å³ç«¯ç‚¹
                res += (len - r +1);

                    mp.erase(n);
                }

            }
        }
        return res;        
    }

class Solution {
    string merge(const string &a, const string &b){
        // return a+ b
        
        // æ³¨æ„æ˜¯ä¸ç­‰äº -1, å› ä¸ºå¯èƒ½æ˜¯ 0
        // ä¸è¦å†™  if(a.find(b) )
        if(a.find(b) != -1){
            return a;
        }

        if(b.find(a) != -1){
            return b;
        }

        for(int i = min(a.size(), b.size()); i>-1; --i){
            if(a.substr(a.size( ) -i ) == b.substr(0, i) ){
                // cout<< a+ b.substr(i)<< '\n';
                return a+ b.substr(i);
            }
        }
        return "";
    }
public:
    string minimumString(string a, string b, string c) {
        string arr[] = {a, b, c};

        string ans = "";
        // æšä¸¾ arr çš„å…¨æ’åˆ—
        int perm[][3] = {   {0, 1, 2}, {0, 2, 1}, {1, 0, 2}, {1, 2, 0}, {2, 0, 1}, {2, 1, 0}};
        for (const auto& p : perm) {
            string s = merge(merge(arr[p[0]], arr[p[1]]), arr[p[2]]);
            if (ans.empty() || s.length() < ans.length() || (s.length() == ans.length() && s < ans)) {
                ans = s;
            }
        }
        return ans;
    }
};
