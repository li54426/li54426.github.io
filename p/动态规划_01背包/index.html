#include<iostream>
#include<vector>
using namespace std;

int main(){
    // the number of things, and the volume of bags
    
    int N,V;
    cin >> N >>V;
    
    vector<int> w(N,0), v(N, 0), dp(V+1, 0);
    
    for(int i =0; i<N; ++i){
        cin>> w[i] >>v[i];
        
    }
    
    //init
    for(int i =w[0];i<= V; ++i){
        dp[i] = v[0];
    }
    
    for(int i =1;i< N; ++i){
        for(int j = V; j>= w[i]; --j){
            dp[j] = max(dp[j- w[i]] + v[i], dp[j]);
            
        }
    }
    
    cout<< dp[V]<< endl;
    return 0;
    
}
